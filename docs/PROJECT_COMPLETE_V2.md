## 💡 Quick Start

```bash
# Build
make build

# Test
make test

# CLI quick start
sdk system health
sdk agent list
sdk task create --description "Explore zone" --priority 150

# Simulation
cargo run --example simulation_demo -- --scenario multi_agent_coordination

# NLP control
echo "Create a high priority task for exploration" | sdk nlp parse
sdk nlp search "explore area" --limit 10

# Mobile app
cd mobile-dashboard
npm install
npm start

# Docker deployment
docker-compose up -d

# Kubernetes deployment
helm install sdk-operator ./kubernetes/operator-chart

# API Gateway
curl http://localhost:8000/api/v1/tasks

# Multi-tenancy
sdk tenant create --name "Acme Corp"
sdk tenant assign --user "user-1" --tenant "tenant-123"

# Blockchain integration
cargo run --example blockchain_demo

# ZKP demo
cargo run --example zkp_demo
```
