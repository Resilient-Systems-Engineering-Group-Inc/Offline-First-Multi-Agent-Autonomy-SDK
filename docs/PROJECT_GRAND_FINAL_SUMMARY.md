## 💡 Quick Start

```bash
# Build
make build

# Test
make test

# CLI quick start
sdk system health
sdk agent list
sdk task create --description "Explore zone" --priority 150

# Event sourcing
cargo run --example event_sourcing_demo -- --stream task-1

# Edge AI inference
cargo run --example edge_ai_infer -- --model ./models/planner.onnx --input image.jpg

# Digital twin
cargo run --example digital_twin -- --entity agent-1 --simulate

# Simulation
cargo run --example simulation_demo -- --scenario multi_agent_coordination

# Performance testing
cargo run --example perf_test -- --load-test basic_api

# NLP control
echo "Create a high priority task for exploration" | sdk nlp parse
sdk nlp search "explore area" --limit 10

# Health check
curl http://localhost:8000/health

# MLOps
cargo run --example mlops_demo -- --train planner_model

# Graph queries
cargo run --example graph_demo -- --query "MATCH (a:Agent) RETURN a"

# Mobile app
cd mobile-dashboard
npm install
npm start

# Docker deployment
docker-compose up -d

# Kubernetes deployment
helm install sdk-operator ./kubernetes/operator-chart
```
